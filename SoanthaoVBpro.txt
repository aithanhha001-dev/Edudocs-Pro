        // Handle document/image file upload processing
        async function handleReferenceUpload(input) {
            const file = input.files[0];
            if (!file) return;

            const name = file.name;
            const ext = name.split('.').pop().toLowerCase();
            
            // Adjust visual states
            document.getElementById('dropzone-prompt').classList.add('hidden');
            document.getElementById('dropzone-success').classList.remove('hidden');
            document.getElementById('ref-file-name').innerText = name;
            document.getElementById('ref-file-status').innerText = "Đang xử lý nạp tài liệu...";
            document.getElementById('ref-file-status').className = "text-[8px] text-amber-600 leading-none mt-0.5";
            document.getElementById('ref-preview-box').classList.add('hidden');

            const iconEl = document.getElementById('ref-icon');
            const iconContainer = document.getElementById('ref-icon-container');
            iconContainer.className = "bg-amber-500 text-white p-1.5 rounded-lg flex items-center justify-center shrink-0 shadow-sm";

            try {
                if (ext === 'pdf') {
                    iconEl.className = "fa-solid fa-file-pdf text-xs";
                    const text = await extractPdfText(file);
                    referenceSource = { type: 'text', data: text, name: name, mimeType: null };
                    showUploadSuccess(name, text.substring(0, 400) + "...");
                } else if (['png', 'jpg', 'jpeg'].includes(ext)) {
                    iconEl.className = "fa-solid fa-file-image text-xs";
                    const base64Data = await fileToBase64(file);
                    // Standard raw base64 extraction
                    const cleanBase64 = base64Data.split(',')[1];
                    referenceSource = { type: 'image', data: cleanBase64, mimeType: file.type, name: name };
                    showUploadSuccess(name, "[Nguồn ảnh trực quan đã sẵn sàng. Trợ lý AI sẽ đọc trực tiếp nội dung trong ảnh này để phản hồi.]");
                } else if (ext === 'txt') {
                    iconEl.className = "fa-solid fa-file-lines text-xs";
                    const text = await readFileAsText(file);
                    referenceSource = { type: 'text', data: text, name: name, mimeType: null };
                    showUploadSuccess(name, text.substring(0, 400) + "...");
                } else {
                    showToast("Định dạng không hỗ trợ", "Chương trình hỗ trợ đính kèm tệp PDF, Văn bản .txt hoặc ảnh chụp tài liệu.", "fa-triangle-exclamation");
                    resetUploadZone();
                }
            } catch (err) {
                console.error("File processing error: ", err);
                showToast("Lỗi nạp tệp", "Không thể trích lọc dữ liệu từ tệp tin đã chọn.", "fa-triangle-exclamation");
                resetUploadZone();
            }
        }

        // Parse and extract PDF pages to text directly inside user's browser
        function extractPdfText(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = async function() {
                    try {
                        const typedarray = new Uint8Array(this.result);
                        const pdf = await pdfjsLib.getDocument(typedarray).promise;
                        let extractedText = "";
                        const maxPages = Math.min(pdf.numPages, 10); // Limit processing to first 10 pages for performance
                        
                        for (let i = 1; i <= maxPages; i++) {
                            const page = await pdf.getPage(i);
                            const textContent = await page.getTextContent();
                            const pageText = textContent.items.map(item => item.str).join(' ');
                            extractedText += `[TRANG HỌC LIỆU ${i}]\n` + pageText + "\n";
                        }
                        resolve(extractedText);
                    } catch (e) {
                        reject(e);
                    }
                };
                reader.onerror = reject;
                reader.readAsArrayBuffer(file);
            });
        }

        // Helper to read plain text files
        function readFileAsText(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = () => resolve(reader.result);
                reader.onerror = reject;
                reader.readAsText(file);
            });
        }

        // Convert files to clean base64 data URLs
        function fileToBase64(file) {
            return new Promise((resolve, reject) => {
                const reader = new FileReader();
                reader.onload = () => resolve(reader.result);
                reader.onerror = reject;
                reader.readAsDataURL(file);
            });
        }

        // Fetch with exponential backoff helper
        async function fetchWithRetry(url, payload, retries = 3, delay = 1000) {
            for (let i = 0; i < retries; i++) {
                try {
                    const response = await fetch(url, {
                        method: 'POST',
                        headers: { 'Content-Type': 'application/json' },
                        body: JSON.stringify(payload)
                    });
                    if (response.ok) return response;
                } catch (e) {
                    if (i === retries - 1) throw e;
                }
                await new Promise(resolve => setTimeout(resolve, delay * Math.pow(2, i)));
            }
        }

            if(!dateVal) return;
            const d = new Date(dateVal);
            document.getElementById('lbl-day').innerText = String(d.getDate()).padStart(2, '0');
            document.getElementById('lbl-month').innerText = String(d.getMonth() + 1).padStart(2, '0');
            document.getElementById('lbl-year').innerText = d.getFullYear();
        }

        // Sync recipients block
        function syncRecipients(text) {
            const lines = text.split('\n');
            const container = document.getElementById('lbl-recipients');
            container.innerHTML = '';
            
            lines.forEach(line => {
                if(line.trim() !== '') {
                    let formattedLine = line.trim();
                    if (!formattedLine.startsWith('-')) {
                        formattedLine = '- ' + formattedLine;
                    }
                    const div = document.createElement('div');
                    div.innerText = formattedLine;
                    container.appendChild(div);
                }
            });
        }

        // Load a specific document template dynamically
        function loadTemplate(templateKey) {
            const data = templates[templateKey];
            if (!data) return;

            toggleDocMode(data.docMode);

            if (data.docMode === 'admin') {
                document.getElementById('lbl-subhead').innerHTML = data.subhead;
                document.getElementById('lbl-school').innerHTML = data.school;
                document.getElementById('lbl-docnumber').innerHTML = data.docnumber;
                document.getElementById('lbl-location').innerHTML = data.location;
                document.getElementById('lbl-signer-title').innerHTML = data.signerTitle;
                document.getElementById('lbl-signer-name').innerHTML = data.signerName;
                
                // Format and load Recipients in DOM
                const recContainer = document.getElementById('lbl-recipients');
                recContainer.innerHTML = '';
                const lines = data.recipients.split('\n');
                lines.forEach(line => {
                    const div = document.createElement('div');
                    div.innerText = line;
                    recContainer.appendChild(div);
                });
            } else {
                document.getElementById('lbl-lesson-school').innerHTML = data.school;
                document.getElementById('lbl-lesson-dept').innerHTML = data.dept;
                document.getElementById('lbl-lesson-week').innerHTML = data.week;
                document.getElementById('lbl-lesson-period').innerHTML = data.period;
                document.getElementById('lbl-lesson-class').innerHTML = data.class;
                document.getElementById('lbl-lesson-prepdate').innerHTML = data.prepdate;
                document.getElementById('lbl-lesson-teacher').innerHTML = data.teacher;
                document.getElementById('lbl-lesson-approver').innerHTML = data.approver;
            }

            document.getElementById('lbl-title').innerHTML = data.title;
            document.getElementById('document-body').innerHTML = data.body;

            // Refresh form fields in the sidebar to match the newly selected template
            populateInputsFromPreview();

            showToast("Đã tải biểu mẫu", `Cấu trúc mẫu "${templateKey === 'giaoan' ? 'Giáo án' : 'Hành chính'}" mới đã được nạp thành công.`, "fa-file-invoice");
        }

                    recipients: document.getElementById('lbl-recipients').innerHTML,

                    // Lesson plan metadata
                    lessonSchool: document.getElementById('lbl-lesson-school').innerText,
                    lessonDept: document.getElementById('lbl-lesson-dept').innerText,
                    lessonWeek: document.getElementById('lbl-lesson-week').innerText,
                    lessonPeriod: document.getElementById('lbl-lesson-period').innerText,
                    lessonClass: document.getElementById('lbl-lesson-class').innerText,
                    lessonPrepdate: document.getElementById('lbl-lesson-prepdate').innerText,
                    lessonTeacher: document.getElementById('lbl-lesson-teacher').innerText,
                    lessonApprover: document.getElementById('lbl-lesson-approver').innerText
                };

                let savedDrafts = JSON.parse(localStorage.getItem('edu_docs_drafts') || '[]');
                if(savedDrafts.length >= 5) {
                    savedDrafts.shift();
                }
                savedDrafts.push(draft);
                localStorage.setItem('edu_docs_drafts', JSON.stringify(savedDrafts));
                
                loadSavedDraftsList();
                showToast("Lưu bản nháp thành công", "Bản nháp hiện tại đã lưu trữ an toàn trong trình duyệt của bạn.", "fa-floppy-disk");
            } catch (e) {
                showToast("Lỗi lưu trữ", "Có lỗi xảy ra khi lưu trữ bản nháp.", "fa-triangle-exclamation");
            }
        }

        // Render drafts history inside sidebar list
        function loadSavedDraftsList() {
            const savedDrafts = JSON.parse(localStorage.getItem('edu_docs_drafts') || '[]');
            const container = document.getElementById('draftContainer');
            const list = document.getElementById('draftsList');
            
            if (savedDrafts.length === 0) {
                container.classList.add('hidden');
                return;
            }

            container.classList.remove('hidden');
            list.innerHTML = '';

            savedDrafts.reverse().forEach(draft => {
                const div = document.createElement('div');
                div.className = "flex justify-between items-center bg-white p-2 rounded border border-slate-200 shadow-sm";
                
                let shortTitle = draft.title.replace(/<br>/g, ' ').substring(0, 30) + '...';
                let tagColor = draft.docMode === 'admin' ? 'bg-blue-100 text-blue-700' : 'bg-purple-100 text-purple-700';
                let tagName = draft.docMode === 'admin' ? 'HC' : 'GA';

                div.innerHTML = `
                    <div class="cursor-pointer hover:text-blue-600 flex-1 min-w-0 mr-2" onclick="recoverDraft('${draft.id}')">
                        <span class="font-semibold text-[11px] truncate block flex items-center gap-1.5" title="${draft.title.replace(/<br>/g, ' ')}">
                            <span class="text-[8px] font-bold px-1 py-0.5 rounded ${tagColor}">${tagName}</span>
                            <span>${shortTitle}</span>
                        </span>
                        <span class="text-[9px] text-slate-400 block">${draft.time}</span>
                    </div>
                    <button onclick="deleteDraft('${draft.id}')" class="text-red-500 hover:bg-red-50 p-1.5 rounded transition shrink-0" title="Xóa bản nháp này">
                        <i class="fa-regular fa-trash-can text-[11px]"></i>
                    </button>
                `;
                list.appendChild(div);
            });
        }

        // Recover selected Draft content into paper canvas
        function recoverDraft(draftId) {
            const savedDrafts = JSON.parse(localStorage.getItem('edu_docs_drafts') || '[]');
            const draft = savedDrafts.find(d => d.id === draftId);
            if (!draft) return;

            toggleDocMode(draft.docMode || 'admin');

            document.getElementById('lbl-title').innerHTML = draft.title;
            document.getElementById('document-body').innerHTML = draft.body;

            if (draft.docMode === 'admin') {
                document.getElementById('lbl-subhead').innerHTML = draft.subhead || "";
                document.getElementById('lbl-school').innerHTML = draft.school || "";
                document.getElementById('lbl-docnumber').innerHTML = draft.docnumber || "";
                document.getElementById('lbl-location').innerHTML = draft.location || "";
                document.getElementById('lbl-signer-title').innerHTML = draft.signerTitle || "";
                document.getElementById('lbl-signer-name').innerHTML = draft.signerName || "";
                document.getElementById('lbl-recipients').innerHTML = draft.recipients || "";
            } else {
                document.getElementById('lbl-lesson-school').innerHTML = draft.lessonSchool || "TRƯỜNG THPT NGUYỄN DU";
                document.getElementById('lbl-lesson-dept').innerHTML = draft.lessonDept || "Tổ chuyên môn: Toán - Tin học";
                document.getElementById('lbl-lesson-week').innerHTML = draft.lessonWeek || "Tuần 1";
                document.getElementById('lbl-lesson-period').innerHTML = draft.lessonPeriod || "Tiết 1";
                document.getElementById('lbl-lesson-class').innerHTML = draft.lessonClass || "Khối 10";
                document.getElementById('lbl-lesson-prepdate').innerHTML = draft.lessonPrepdate || "18/10/2026";
                document.getElementById('lbl-lesson-teacher').innerHTML = draft.lessonTeacher || "Nguyễn Văn Thắng";
                document.getElementById('lbl-lesson-approver').innerHTML = draft.lessonApprover || "Trần Thị Lan";
            }

            populateInputsFromPreview();
            showToast("Phục hồi thành công", `Đã khôi phục bản nháp học đường ngày ${draft.time}`, "fa-clock-rotate-left");
        }

        // Delete specific draft item
        function deleteDraft(draftId) {
            let savedDrafts = JSON.parse(localStorage.getItem('edu_docs_drafts') || '[]');
            savedDrafts = savedDrafts.filter(d => d.id !== draftId);
            localStorage.setItem('edu_docs_drafts', JSON.stringify(savedDrafts));
            loadSavedDraftsList();
            showToast("Đã xóa bản nháp", "Đã gỡ bản nháp đã chọn khỏi bộ nhớ trình duyệt.", "fa-trash-can");
        }

        // Print/Export to PDF Engine using OS native print
        function exportToPDF() {
            // Turn off margins representation line so it looks premium and clean
            const paper = document.getElementById('paper');
            paper.classList.remove('show-margins');
            document.getElementById('marginBtn').classList.remove('bg-blue-600', 'text-white');
            document.getElementById('marginBtn').classList.add('bg-slate-800', 'text-slate-300');

            resetZoom();
            window.print();
        }

        // Custom modern non-blocking Toast feedback
        let toastTimeout;
        function showToast(title, msg, iconClass = "fa-bell") {
            const toast = document.getElementById('toast');
            const icon = document.getElementById('toast-icon');
            
            document.getElementById('toast-title').innerText = title;
            document.getElementById('toast-message').innerText = msg;
            icon.innerHTML = `<i class="fa-solid ${iconClass}"></i>`;

            clearTimeout(toastTimeout);
            
            toast.classList.remove('translate-y-24', 'opacity-0');
            toast.classList.add('translate-y-0', 'opacity-100');

            toastTimeout = setTimeout(() => {
                toast.classList.remove('translate-y-0', 'opacity-100');
                toast.classList.add('translate-y-24', 'opacity-0');
            }, 4000);
        }
